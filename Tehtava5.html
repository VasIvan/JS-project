function myFunction(){
var lampotila = document.getElementById('lampo').value;

}
else if (lampotila >= 12 && lampotila <= 24 ){
alert(lampotila + ' astetta, Lämmintä riittää!' );
}
else if (lampotila >= 1 && lampotila <= 11 ){
alert(lampotila + ' astetta, Vähän lämmintä!' );
}
else if (lampotila == 0 ){
alert(lampotila + ' astetta, Mittari nollassa!' );
} else if (lampotila <= -1 && lampotila >= -12 ){
alert(lampotila + ' astetta, Pikku pakkasta!' );
}
else if (lampotila <= -13 ){
alert(lampotila + ' astetta, Nyt on kova pakkanen!' );
}
else {
alert('Tarkista annettu lämpötila!' );
}
}
